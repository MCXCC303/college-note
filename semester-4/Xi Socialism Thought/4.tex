\begin{itemize}
    \item 经济和科技大幅提升
    \item 国家治理现代化
    \item 文化和社会建设
    \item 生态文明建设
    \item 民生与共同富裕
    \item 国防安全
    \item 对外开放
\end{itemize}

\end{itemize}
