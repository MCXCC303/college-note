\begin{itemize}
    \item motobike too noisy
    \item narrow way
    \item cars too many
    \item traffic lights
    \item subway too busy
    \item slow buses
    \item \ldots 
    
\end{itemize}

    What has the residents said:

        \item 5, too much noise: 60\%
        \item 8, on-street parking: 45\%
        \item 7, unreliable bus service: 40\%
    \end{itemize}
\end{notation}
